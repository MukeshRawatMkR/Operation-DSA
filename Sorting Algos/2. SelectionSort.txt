//Biggest ele at the end after each iteration.
import java.util.* ;
import java.io.*; 
public class Solution {

public static void swap(int[] arr, int i, int j){
	int temp=arr[i];
	arr[i]=arr[j];
	arr[j]=temp;
}

	public static void selectionSort(int arr[], int n) {
		//6 =+% `

		for(int i=0;i<arr.length-1;i++){
			int minIndex=i;
			for(int j=i+1;j<arr.length;j++){
				if(arr[minIndex]>arr[j]){
					minIndex=j;
				}
			}
			if(minIndex != i) //to avoid unnecceasary swapping.
			swap(arr, minIndex, i);
				}
	}
}

//NOTES:
/*
Complexity
	Worst / Average TC: O(n^2)
	Best TC: O(n^2)
	Space Complexity: O(1)
Properties
	In-place: ✅ Yes
	Stable: Not stable (because swapping can disturb equal elements)
*/
